//circular concolution
/*Circular convolution is mostly used for continuous signal. 
In circular convolution length of both input [m(n)] + [n(n)] signal is same as the signal length of output signal*/

clc;
clf;
clear all;
xn=input("Enter x[n]: ");
subplot(3,1,1);
plot2d3(xn);
hn=input("Enter h[n]: ");
subplot(3,1,2);
plot2d3(hn);
x_new=[];
for i = 1:length(xn)
  x_new=[xn;x_new];
  xn=[xn($) xn(1:$-1)];
end
yn=x_new'*hn';
subplot(3,1,3);
plot2d3(yn);
disp(yn);

//[1,2,3,4] \ [1,1,1,1]
